const watercolors = document.getElementById("watercolors");

const watercolorcount = 4;
for (let i = 1; i <= watercolorcount; i++) {
    const wrapper = document.createElement("div");
    wrapper.className = "photo-box";

    const img = document.createElement("img");
    img.src = `images/watercolor${i}.jpg`;
    img.alt = `Painting ${i}`;
    wrapper.appendChild(img);
    gallery.appendChild(wrapper);
}
